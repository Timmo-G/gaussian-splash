```python
import numpy as np
def numerical_partial(f, x, y, var='x', eps=1e-5):
    if var == 'x':
        return (f(x+eps, y) - f(x-eps, y)) / (2*eps)
    else:
        return (f(x, y+eps) - f(x, y-eps)) / (2*eps)
```

### P2. 梯度检验工具

实现一个通用的梯度检验函数：

```python
def gradient_check(f, grad_f, params, eps=1e-5, tolerance=1e-4):
    """
    检验解析梯度是否正确
    f: 标量函数
    grad_f: 梯度函数
    params: 参数向量
    返回: (是否通过, 最大相对误差)
    """
    analytical = grad_f(params)
    numerical = numerical_gradient(f, params, eps)
    
    rel_error = np.abs(analytical - numerical) / (np.abs(numerical) + 1e-8)
    passed = np.max(rel_error) < tolerance
    return passed, np.max(rel_error)
```

```python
def l1_loss(pred, target):

def dssim_loss(pred, target):
    """D-SSIM = (1 - SSIM) / 2"""
    from skimage.metrics import structural_similarity as ssim
    return (1 - ssim(pred, target, data_range=1.0)) / 2
```

### P5. 学习率调度

3DGS使用指数衰减学习率调度：

```python
# train.py 中的学习率调度
def get_expon_lr_func(lr_init, lr_final, lr_delay_steps=0,
                      lr_delay_mult=1.0, max_steps=1000000):
    """指数衰减学习率"""
    def helper(step):
        if step < 0 or (lr_init == 0.0 and lr_final == 0.0):
            return 0.0
        if lr_delay_steps > 0:
            delay_rate = lr_delay_mult + (1 - lr_delay_mult) * np.sin(
                0.5 * np.pi * np.clip(step / lr_delay_steps, 0, 1))
        else:
            delay_rate = 1.0
        t = np.clip(step / max_steps, 0, 1)
        log_lerp = np.exp(np.log(lr_init) * (1-t) + np.log(lr_final) * t)
        return delay_rate * log_lerp
    return helper
```

2. 解释为什么使用指数衰减而不是线性衰减
3. 对比：固定学习率 vs 指数衰减，在简单优化问题上的效果
